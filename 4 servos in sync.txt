#include <Servo.h>

Servo servo_9;
Servo servo_8;
Servo servo_10;
Servo servo_11;

void setup() {
  servo_9.attach(9, 500, 2500);
  servo_8.attach(8, 500, 2500);
  servo_10.attach(10, 500, 2500);
  servo_11.attach(11, 500, 2500);

  // Sweep from 0 to 180
  for (int pos = 0; pos <= 180; pos += 1) {
    servo_9.write(pos);
    servo_8.write(pos);
    servo_10.write(pos);
    servo_11.write(pos);
    delay(10); // ~1.8 sec for full sweep
  }

  // Sweep back from 180 to 0
  for (int pos = 180; pos >= 0; pos -= 1) {
    servo_9.write(pos);
    servo_8.write(pos);
    servo_10.write(pos);
    servo_11.write(pos);
    delay(10); // ~1.8 sec for return
  }

  // Hold at 90Â°
  servo_9.write(90);
  servo_8.write(90);
  servo_10.write(90);
  servo_11.write(90);
}

void loop() {
  // Do nothing after the sweep
}
